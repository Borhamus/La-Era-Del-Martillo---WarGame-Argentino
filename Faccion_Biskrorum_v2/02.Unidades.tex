\begin{loreblock}

\end{loreblock}

% ==========================================
% UNIDAD 1: CAZA SOMBRA-KIEMÖN
% ==========================================
\newpage
\fichaFrente{Caza Sombra-Kiemön}{12}{Nave Ligera / DPS Rango}{40cm}{+1}{+5}{+4}{+1}{+6}

\newpage
\fichaDorso{Caza Sombra-Kiemön}{
    \item \textbf{Aceleración Abisal:} Si esta unidad se movió más de 20cm este turno, gana \textbf{+2 REF} hasta el inicio de su próxima activación.
    \item \textbf{Rayo del Vacío:} Ataque de Rango que \textbf{ignora completamente la RD} del objetivo. Coste: 1 PA + 1 PE.
    \item \textbf{Protocolo Suicida:} Al ser eliminada, explota causando \textbf{1d6 de daño directo} a todas las unidades (aliadas y enemigas) en 5cm. La miniatura queda en el tablero como escombro.
}{
    \textit{``No es una construcción de metal, sino de voluntad pura. El Sombra-Kiemön es el primer suspiro del Padre al emerger del mar. Dentro de su casco vive el alma de un Biugrom que eligió este sacrificio. Dura horas. Destruye todo en ese tiempo.''}
}

% ==========================================
% UNIDAD 2: NÓDULOS DE CHOQUE
% ==========================================
\newpage
\fichaFrente{Nódulos de Choque}{5}{Enjambre / Obstrucción}{15cm}{+3}{+2}{+2}{+2}{+3}

\newpage
\fichaDorso{Nódulos de Choque}{
    \item \textbf{Explosión de Biomasa:} Cuando esta unidad es retirada del juego, hace estallar su biomasa causando \textbf{1d4 de daño} en 3cm de radio. No afecta a otras unidades Biskrorum.
    \item \textbf{Manto Viviente:} El Nódulo puede sacrificarse antes de que un aliado a 2cm reciba daño. El Nódulo absorbe \textbf{hasta 3 de daño} del ataque y luego es eliminado.
    \item \textbf{Señal de Enjambre:} Si 3 o más Nódulos están dentro de 8cm entre sí, todos ganan \textbf{+1 MELEE}.
}{
    \textit{``Balas vivientes. Nacen de huevos abisales y regresan al Vacío cuando su biomasa se agota. No tienen nombre. No tienen memoria. Tienen propósito.''}
}
